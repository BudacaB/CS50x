CREATE TABLE IF NOT EXISTS 'stocks' ('userid' INTEGER NOT NULL, 'symbol' TEXT NOT NULL, 'name' TEXT NOT NULL, 'shares' INTEGER NOT NULL, FOREIGN KEY(userid) REFERENCES users(id));

CREATE TABLE IF NOT EXISTS 'history' ('userid' INTEGER NOT NULL, 'symbol' TEXT NOT NULL, 'shares' INTEGER NOT NULL, 'price' NUMERIC NOT NULL, 'transacted' TEXT NOT NULL, FOREIGN KEY(userid) REFERENCES users(id));
